MEPH.define('MEPH.qrcode.version.ECB', {
    initialize: function (count, dataCodewords) {
        var me = this;
        me.count = count;
        me.dataCodewords = dataCodewords;
        Object.defineProperty(me, 'Count', {
            get: function () {
                return me.count;
            }
        });
        Object.defineProperty(me, 'DataCodewords', {
            get: function () {
                return this.dataCodewords;
            }
        });
    }
});
